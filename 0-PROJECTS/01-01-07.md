## Key Concepts Explanation

### Model Context Protocol (MCP)
Advanced communication framework enabling AI models to intelligently interact with healthcare information systems, Electronic Health Records (EHR), and medical databases while maintaining strict privacy controls and contextual understanding of medical terminology, patient care workflows, and clinical decision-making processes.

### Medical Records Management
Comprehensive digital patient information system that stores, organizes, and manages clinical data including patient demographics, medical history, diagnostic results, treatment plans, medications, and care coordination across multiple healthcare providers and specialties.

### FHIR Standards (Fast Healthcare Interoperability Resources)
International standard for healthcare data exchange that defines how healthcare information can be exchanged between different computer systems regardless of how it is stored, enabling seamless interoperability between hospitals, clinics, pharmacies, and other healthcare entities.

### Patient Data Analysis
Advanced analytics and AI-powered insights derived from clinical data to identify patterns, predict health outcomes, suggest treatment optimizations, detect potential complications, and support evidence-based medical decision-making while preserving patient privacy.

### Compliance Monitoring
Automated systems that ensure healthcare operations adhere to regulatory requirements including HIPAA, GDPR, FDA guidelines, and medical standards through continuous monitoring, audit trails, access controls, and compliance reporting mechanisms.

### HIPAA (Health Insurance Portability and Accountability Act)
Federal law establishing national standards for protecting patient health information privacy and security, requiring healthcare entities to implement administrative, physical, and technical safeguards to protect patients' medical records and personal health information.

class EncryptionManager:
    """HIPAA-compliant encryption manager for patient data"""
    
    def __init__(self, master_key: bytes = None):
        if master_key is None:
            master_key = Fernet.generate_key()
        self.cipher_suite = Fernet(master_key)
        self.master_key = master_key
    
    def encrypt_data(self, data: str) -> bytes:
        """Encrypt sensitive patient data"""
        try:
            return self.cipher_suite.encrypt(data.encode('utf-8'))
        except Exception as e:
            logger.error(f"Encryption failed: {e}")
            raise
    
    def decrypt_data(self, encrypted_data: bytes) -> str:
        """Decrypt patient data"""
        try:
            return self.cipher_suite.decrypt(encrypted_data).decode('utf-8')
        except Exception as e:
            logger.error(f"Decryption failed: {e}")
            raise
    
    def generate_data_hash(self, data: str) -> str:
        """Generate hash for data integrity verification"""
        return hashlib.sha256(data.encode('utf-8')).hexdigest()
    
    def verify_data_integrity(self, data: str, stored_hash: str) -> bool:
        """Verify data hasn't been tampered with"""
        return self.generate_data_hash(data) == stored_hash

class FHIRManager:
    """FHIR standard implementation for healthcare interoperability"""
    
    def __init__(self, encryption_manager: EncryptionManager):
        self.encryption_manager = encryption_manager
        self.fhir_version = "R4"
    
    async def create_patient_resource(self, patient_data: PatientData) -> Patient:
        """Create FHIR Patient resource"""
        try:
            patient = Patient()
            
            # Basic demographics
            patient.id = patient_data.patient_id
            patient.active = True
            
            # Name
            if patient_data.demographics.get('name'):
                from fhir.resources.humanname import HumanName
                name = HumanName()
                name_parts = patient_data.demographics['name'].split()
                name.given = name_parts[:-1] if len(name_parts) > 1 else [name_parts[0]]
                name.family = name_parts[-1] if len(name_parts) > 1 else ""
                patient.name = [name]
            
            # Gender
            if patient_data.demographics.get('gender'):
                patient.gender = patient_data.demographics['gender'].lower()
            
            # Birth date
            if patient_data.demographics.get('birth_date'):
                from fhir.resources.fhirtypes import Date
                patient.birthDate = Date(patient_data.demographics['birth_date'])
            
            # Contact information
            if patient_data.demographics.get('phone') or patient_data.demographics.get('email'):
                from fhir.resources.contactpoint import ContactPoint
                telecoms = []
                
                if patient_data.demographics.get('phone'):
                    phone = ContactPoint()
                    phone.system = "phone"
                    phone.value = patient_data.demographics['phone']
                    telecoms.append(phone)
                
                if patient_data.demographics.get('email'):
                    email = ContactPoint()
                    email.system = "email"
                    email.value = patient_data.demographics['email']
                    telecoms.append(email)
                
                patient.telecom = telecoms
            
            return patient
            
        except Exception as e:
            logger.error(f"FHIR Patient resource creation failed: {e}")
            raise
    
    async def create_observation_resource(self, patient_id: str,
                                        observation_data: Dict[str, Any]) -> Observation:
        """Create FHIR Observation resource"""
        try:
            observation = Observation()
            observation.id = str(uuid.uuid4())
            observation.status = "final"
            
            # Subject (patient reference)
            from fhir.resources.reference import Reference
            observation.subject = Reference(reference=f"Patient/{patient_id}")
            
            # Code (what was observed)
            from fhir.resources.codeableconcept import CodeableConcept
            from fhir.resources.coding import Coding
            
            code = CodeableConcept()
            coding = Coding()
            coding.system = "http://loinc.org"
            coding.code = observation_data.get('loinc_code', '55284-4')
            coding.display = observation_data.get('name', 'Observation')
            code.coding = [coding]
            observation.code = code
            
            # Value
            if observation_data.get('value'):
                from fhir.resources.quantity import Quantity
                value_quantity = Quantity()
                value_quantity.value = float(observation_data['value'])
                value_quantity.unit = observation_data.get('unit', '')
                observation.valueQuantity = value_quantity
            
            # Effective date
            from fhir.resources.fhirtypes import DateTime
            observation.effectiveDateTime = DateTime(
                observation_data.get('date', datetime.now().isoformat())
            )
            
            return observation
            
        except Exception as e:
            logger.error(f"FHIR Observation resource creation failed: {e}")
            raise

    async def bundle_resources(self, resources: List[Any]) -> Bundle:

        try:

    async def _create_fhir_resources(self, patient_data: PatientData):
        """Create FHIR resources for patient data"""
        try:
            # Create Patient resource
            patient_resource = await self.fhir_manager.create_patient_resource(patient_data)
            
            # Create Observation resources for vital signs
            observations = []
            for vital_name, vital_value in patient_data.vital_signs.items():
                if vital_value:
                    observation_data = {
                        'name': vital_name,
                        'value': vital_value,
                        'unit': self._get_vital_unit(vital_name),
                        'date': datetime.now().isoformat()
                    }
                    observation = await self.fhir_manager.create_observation_resource(
                        patient_data.patient_id, observation_data
                    )
                    observations.append(observation)
            
            # Create Condition resources for diagnoses
            conditions = []
            for diagnosis in patient_data.diagnoses:
                condition = await self.fhir_manager.create_condition_resource(
                    patient_data.patient_id, diagnosis
                )
                conditions.append(condition)
            
            # Bundle all resources
            all_resources = [patient_resource] + observations + conditions
            bundle = await self.fhir_manager.bundle_resources(all_resources)
            
            # Store FHIR bundle
            async with self.session_factory() as session:
                fhir_record = FHIRResource(
                    id=str(uuid.uuid4()),
                    resource_type="Bundle",
                    resource_id=bundle.id,
                    patient_id=patient_data.patient_id,
                    resource_data=bundle.dict()
                )
                session.add(fhir_record)
                await session.commit()
            
        except Exception as e:
            logger.error(f"FHIR resource creation failed: {e}")

    async def generate_clinical_note(self, patient_id: str, note_type: str,

        """Generate AI-powered clinical note"""
        try:
            # Get patient data for context
            patient_data = await self._get_patient_data(patient_id)
            
            note_prompt = f"""
            Generate a professional {note_type} clinical note for patient {patient_id}:
            
            Patient Context:
            - Demographics: {json.dumps(patient_data.demographics, indent=2) if patient_data else 'N/A'}
            - Current Medications: {json.dumps(patient_data.current_medications, indent=2) if patient_data else 'N/A'}

    async def _get_patient_data(self, patient_id: str) -> Optional[PatientData]:
        """Retrieve and decrypt patient data"""
        try:
            async with self.session_factory() as session:
                result = await session.execute(
                    "SELECT * FROM patient_records WHERE patient_id = ?", (patient_id,)
                )
                record = result.fetchone()
                
                if not record:
                    return None
                
                # Decrypt data
                decrypted_data = self.encryption_manager.decrypt_data(record.encrypted_data)
                patient_dict = json.loads(decrypted_data)
                
                return PatientData(
                    patient_id=patient_id,
                    demographics=patient_dict.get('demographics', {}),
                    medical_history=patient_dict.get('medical_history', []),
                    current_medications=patient_dict.get('current_medications', []),
                    allergies=patient_dict.get('allergies', []),
                    vital_signs=patient_dict.get('vital_signs', {}),
                    lab_results=patient_dict.get('lab_results', []),
                    diagnoses=patient_dict.get('diagnoses', []),
                    treatment_plan=patient_dict.get('treatment_plan', {})
                )
                
        except Exception as e:
            logger.error(f"Patient data retrieval failed: {e}")
            return None
    
    async def analyze_patient_care(self, patient_id: str) -> List[ClinicalInsight]:
        """Comprehensive analysis of patient care"""
        try:
            # Get patient data
            patient_data = await self._get_patient_data(patient_id)
            if not patient_data:
                return []
            
            # Get clinical notes
            async with self.session_factory() as session:
                result = await session.execute(
                    "SELECT encrypted_content FROM clinical_notes WHERE patient_id = ? ORDER BY created_at DESC LIMIT 5",
                    (patient_id,)
                )
                note_records = result.fetchall()
                
                notes = []
                for record in note_records:
                    decrypted_note = self.encryption_manager.decrypt_data(record.encrypted_content)
                    notes.append(decrypted_note)
            
            # Generate insights
            insights = await self.nlp_processor.analyze_clinical_notes(notes)
            
            # Add patient-specific insights
            patient_insights = await self._generate_patient_insights(patient_data)
            insights.extend(patient_insights)
            
            return insights
            
        except Exception as e:
            logger.error(f"Patient care analysis failed: {e}")
            return []
    
    async def _generate_patient_insights(self, patient_data: PatientData) -> List[ClinicalInsight]:
        """Generate patient-specific clinical insights"""
        insights = []
        
        try:
            # Drug interaction check
            if len(patient_data.current_medications) > 1:
                drug_interaction_prompt = f"""
                Check for potential drug interactions:
                Medications: {json.dumps(patient_data.current_medications, indent=2)}
                Patient Allergies: {', '.join(patient_data.allergies)}
                
                Identify any potential interactions or contraindications.
                """
                
                interaction_analysis = await self.nlp_processor.llm.apredict(drug_interaction_prompt)
                
                if "interaction" in interaction_analysis.lower():
                    insights.append(ClinicalInsight(
                        insight_type="Drug Interaction Alert",
                        description="Potential drug interactions detected",
                        confidence=85.0,
                        supporting_evidence=[interaction_analysis],
                        recommendations=["Review medication list with pharmacist"],
                        risk_level="medium",
                        relevant_guidelines=["Drug interaction guidelines"]
                    ))
            
            # Vital signs analysis
            if patient_data.vital_signs:
                vital_concerns = []
                
                # Check blood pressure
                systolic = patient_data.vital_signs.get('blood_pressure_systolic')
                diastolic = patient_data.vital_signs.get('blood_pressure_diastolic')
                
                if systolic and systolic > 140:
                    vital_concerns.append("Elevated systolic blood pressure")
                if diastolic and diastolic > 90:
                    vital_concerns.append("Elevated diastolic blood pressure")
                
                # Check heart rate
                heart_rate = patient_data.vital_signs.get('heart_rate')
                if heart_rate and (heart_rate > 100 or heart_rate < 60):
                    vital_concerns.append("Abnormal heart rate")
                
                if vital_concerns:
                    insights.append(ClinicalInsight(
                        insight_type="Vital Signs Alert",
                        description=f"Abnormal vital signs detected: {', '.join(vital_concerns)}",
                        confidence=90.0,
                        supporting_evidence=vital_concerns,
                        recommendations=["Monitor vital signs closely", "Consider cardiovascular assessment"],
                        risk_level="medium",
                        relevant_guidelines=["Hypertension management guidelines"]
                    ))
            
            return insights
            
        except Exception as e:
            logger.error(f"Patient insights generation failed: {e}")
            return []
