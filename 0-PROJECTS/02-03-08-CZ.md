class LegalDatabase:
    """Spr√°vce pr√°vn√≠ datab√°ze s vektorov√Ωm vyhled√°v√°n√≠m"""
    
    def __init__(self, db_path: str = "./legal_db"):

            openai_api_key=os.getenv("OPENAI_API_KEY")

            chunk_size=1000,
            chunk_overlap=200

        self._initialize_database()
    
    def _initialize_database(self):
        """Inicializace datab√°ze precedent≈Ø"""
        try:
            self.vectorstore = Chroma(
                persist_directory=self.db_path,
                embedding_function=self.embeddings
            )
        except Exception as e:
            print(f"Chyba p≈ôi inicializaci datab√°ze: {e}")
            self._create_sample_data()
    
    def _create_sample_data(self):

        sample_cases = [

                "facts": "≈Ωalovan√Ω nevr√°til zap≈Øjƒçen√© vozidlo ve smluven√© lh≈Øtƒõ...",
                "outcome": "≈Ωaloba byla √∫spƒõ≈°n√°, p≈ôizn√°na n√°hrada ≈°kody"

            {

                "court": "Nejvy≈°≈°√≠ soud ƒåR",
                "legal_principle": "Odpovƒõdnost za vady zbo≈æ√≠",
                "facts": "Dodan√© zbo≈æ√≠ vykazovalo skryt√© vady...",
                "outcome": "Kupuj√≠c√≠ m√° pr√°vo na vr√°cen√≠ kupn√≠ ceny"
            }
        ]
        
        documents = []
        metadatas = []
        
        for case in sample_cases:
            text = f"{case['case_name']}: {case['facts']} {case['legal_principle']}"
            documents.append(text)
            metadatas.append(case)
        
        self.vectorstore = Chroma.from_texts(
            texts=documents,
            embedding=self.embeddings,
            metadatas=metadatas,
            persist_directory=self.db_path
        )
    
    async def search_precedents(self, query: str, k: int = 5) -> List[Precedent]:
        """Vyhled√°n√≠ relevantn√≠ch precedent≈Ø"""
        try:
            results = self.vectorstore.similarity_search_with_score(query, k=k)
            precedents = []
            
            for doc, score in results:
                precedent = Precedent(
                    case_name=doc.metadata.get('case_name', 'Nezn√°m√Ω'),
                    court=doc.metadata.get('court', 'Nezn√°m√Ω soud'),
                    date=datetime.now(),
                    legal_principle=doc.metadata.get('legal_principle', ''),
                    facts_summary=doc.metadata.get('facts', ''),
                    outcome=doc.metadata.get('outcome', ''),
                    relevance_score=1.0 - score
                )
                precedents.append(precedent)
            
            return precedents
        except Exception as e:
            print(f"Chyba p≈ôi vyhled√°v√°n√≠ precedent≈Ø: {e}")
            return []

class PrecedentAnalyst(Agent):
    """Agent pro anal√Ωzu precedent≈Ø"""
    
    def __init__(self, legal_db: LegalDatabase):
        self.legal_db = legal_db
        super().__init__(
            role="Analytik precedent≈Ø",
            goal="Naj√≠t a analyzovat relevantn√≠ soudn√≠ precedenty",
            backstory="Jsem specializovan√Ω AI agent s expert√≠zou v anal√Ωze soudn√≠ch rozhodnut√≠",
            verbose=True,
            allow_delegation=False,
            llm=ChatOpenAI(temperature=0.1)
        )
    
    async def analyze_precedents(self, case: LegalCase) -> Dict[str, Any]:
        """Anal√Ωza precedent≈Ø pro dan√Ω p≈ô√≠pad"""
        query = f"{' '.join(case.legal_issues)} {case.facts[:500]}"
        precedents = await self.legal_db.search_precedents(query)
        
        analysis = {
            "relevant_precedents": [],
            "legal_principles": [],
            "success_probability": 0.0
        }
        
        for precedent in precedents:
            analysis["relevant_precedents"].append({
                "case_name": precedent.case_name,
                "court": precedent.court,
                "principle": precedent.legal_principle,
                "relevance": precedent.relevance_score
            })
            
            if precedent.legal_principle not in analysis["legal_principles"]:
                analysis["legal_principles"].append(precedent.legal_principle)
        
        # Odhad pravdƒõpodobnosti √∫spƒõchu na z√°kladƒõ precedent≈Ø
        if precedents:
            success_rate = sum(1 for p in precedents if "√∫spƒõ≈°n" in p.outcome.lower()) / len(precedents)
            analysis["success_probability"] = success_rate
        
        return analysis

class ArgumentGenerator(Agent):
    """Agent pro generov√°n√≠ pr√°vn√≠ch argument≈Ø"""
    
    def __init__(self):
        super().__init__(
            role="Gener√°tor argument≈Ø",

            backstory="Specializuji se na tvorbu logick√Ωch a pr√°vnƒõ podlo≈æen√Ωch argument≈Ø",
            verbose=True,
            allow_delegation=False,
            llm=ChatOpenAI(temperature=0.3)
        )
    
    def generate_arguments(self, case: LegalCase, precedent_analysis: Dict[str, Any]) -> Dict[str, List[str]]:
        """Generov√°n√≠ argument≈Ø pro ≈æalobce a ≈æalovan√©ho"""
        
        arguments = {
            "plaintiff_arguments": [],
            "defendant_arguments": [],
            "legal_basis": []
        }
        
        # Argumenty ≈æalobce
        for principle in precedent_analysis["legal_principles"]:
            arguments["plaintiff_arguments"].append(
                f"Na z√°kladƒõ pr√°vn√≠ho principu '{principle}' m√° ≈æalobce n√°rok na uplatnƒõn√≠ sv√Ωch po≈æadavk≈Ø."
            )
        
        # Argumenty ≈æalovan√©ho
        arguments["defendant_arguments"].append(
            "Skutkov√Ω stav p≈ô√≠padu se odli≈°uje od citovan√Ωch precedent≈Ø."

        arguments["defendant_arguments"].append(
            "≈Ωalobn√≠ n√°roky jsou pr√°vnƒõ ned≈Øvodn√© a fakticky nepodlo≈æen√©."
        )
        
        # Pr√°vn√≠ z√°klad
        arguments["legal_basis"] = precedent_analysis["legal_principles"]
        
        return arguments

class StrategicPlanner(Agent):
    """Agent pro strategick√© pl√°nov√°n√≠"""
    
    def __init__(self):
        super().__init__(
            role="Strategick√Ω pl√°novaƒç",
            goal="Vytvo≈ôit komplexn√≠ strategii veden√≠ p≈ô√≠padu",
            backstory="M√°m rozs√°hl√© zku≈°enosti se strategick√Ωm pl√°nov√°n√≠m pr√°vn√≠ch p≈ô√≠pad≈Ø",
            verbose=True,
            allow_delegation=False,
            llm=ChatOpenAI(temperature=0.2)
        )
    
    def create_strategy(self, case: LegalCase, precedent_analysis: Dict[str, Any], 
                      arguments: Dict[str, List[str]]) -> Dict[str, Any]:
        """Vytvo≈ôen√≠ strategick√©ho pl√°nu"""
        
        strategy = {
            "recommended_approach": "",
            "key_arguments": [],
            "potential_challenges": [],
            "success_probability": precedent_analysis["success_probability"],
            "timeline": [],
            "evidence_requirements": []
        }
        
        # Doporuƒçen√Ω p≈ô√≠stup
        if precedent_analysis["success_probability"] > 0.7:
            strategy["recommended_approach"] = "Agresivn√≠ p≈ô√≠stup s d≈Ørazem na siln√© precedenty"
        elif precedent_analysis["success_probability"] > 0.4:
            strategy["recommended_approach"] = "Vyv√°≈æen√Ω p≈ô√≠stup s mo≈ænost√≠ sm√≠ru"
        else:
            strategy["recommended_approach"] = "Obrann√Ω p≈ô√≠stup s minimalizac√≠ rizik"
        
        # Kl√≠ƒçov√© argumenty
        strategy["key_arguments"] = arguments["plaintiff_arguments"][:3]
        
        # Potenci√°ln√≠ v√Ωzvy
        strategy["potential_challenges"] = [
            "Mo≈æn√© protiargumenty protivn√≠ strany",
            "Nedostatek p≈ô√≠m√Ωch precedent≈Ø",
            "Slo≈æitost d≈Økazn√≠ situace"
        ]
        
        # ƒåasov√Ω pl√°n
        strategy["timeline"] = [
            "T√Ωden 1-2: Shrom√°≈ædƒõn√≠ d≈Økaz≈Ø",
            "T√Ωden 3-4: P≈ô√≠prava ≈æaloby",
            "T√Ωden 5-8: P≈ôedbƒõ≈æn√© jedn√°n√≠",
            "T√Ωden 9-12: Hlavn√≠ l√≠ƒçen√≠"
        ]
        
        return strategy

class LegalCaseStrategizer:
    """Hlavn√≠ t≈ô√≠da pro koordinaci multi-agent syst√©mu"""
    
    def __init__(self):
        self.legal_db = LegalDatabase()
        self.precedent_analyst = PrecedentAnalyst(self.legal_db)
        self.argument_generator = ArgumentGenerator()
        self.strategic_planner = StrategicPlanner()
    
    async def analyze_case(self, case: LegalCase) -> Dict[str, Any]:
        """Kompletn√≠ anal√Ωza p≈ô√≠padu"""
        try:
            print(f"Zahajuji anal√Ωzu p≈ô√≠padu: {case.title}")
            
            # 1. Anal√Ωza precedent≈Ø
            print("Analyzuji precedenty...")
            precedent_analysis = await self.precedent_analyst.analyze_precedents(case)
            
            # 2. Generov√°n√≠ argument≈Ø
            print("Generuji argumenty...")
            arguments = self.argument_generator.generate_arguments(case, precedent_analysis)
            
            # 3. Strategick√© pl√°nov√°n√≠
            print("Vytv√°≈ô√≠m strategii...")
            strategy = self.strategic_planner.create_strategy(case, precedent_analysis, arguments)
            
            # Kompletn√≠ anal√Ωza
            complete_analysis = {
                "case_info": {
                    "title": case.title,
                    "type": case.case_type.value,
                    "issues": case.legal_issues
                },
                "precedent_analysis": precedent_analysis,
                "arguments": arguments,
                "strategy": strategy,
                "timestamp": datetime.now().isoformat()
            }
            
            return complete_analysis
            
        except Exception as e:
            print(f"Chyba p≈ôi anal√Ωze p≈ô√≠padu: {e}")
            return {"error": str(e)}
    
    def generate_report(self, analysis: Dict[str, Any]) -> str:
        """Generov√°n√≠ detailn√≠ zpr√°vy"""
        report = f"""
# PR√ÅVN√ç STRATEGICK√Å ANAL√ùZA

## Z√ÅKLADN√ç INFORMACE
- **N√°zev p≈ô√≠padu:** {analysis['case_info']['title']}
- **Typ p≈ô√≠padu:** {analysis['case_info']['type']}
- **Pr√°vn√≠ ot√°zky:** {', '.join(analysis['case_info']['issues'])}

## ANAL√ùZA PRECEDENT≈Æ
- **Poƒçet relevantn√≠ch precedent≈Ø:** {len(analysis['precedent_analysis']['relevant_precedents'])}
- **Pravdƒõpodobnost √∫spƒõchu:** {analysis['precedent_analysis']['success_probability']:.2%}

### Kl√≠ƒçov√© precedenty:
"""
        
        for precedent in analysis['precedent_analysis']['relevant_precedents'][:3]:
            report += f"- **{precedent['case_name']}** ({precedent['court']})\n"
            report += f"  Relevance: {precedent['relevance']:.2f}\n\n"
        
        report += f"""
## ARGUMENTAƒåN√ç STRATEGIE

### Argumenty ≈æalobce:
"""
        for arg in analysis['arguments']['plaintiff_arguments']:
            report += f"- {arg}\n"
        
        report += f"""
### Argumenty ≈æalovan√©ho:
"""
        for arg in analysis['arguments']['defendant_arguments']:
            report += f"- {arg}\n"
        
        report += f"""
## STRATEGICK√â DOPORUƒåEN√ç
- **P≈ô√≠stup:** {analysis['strategy']['recommended_approach']}
- **Pravdƒõpodobnost √∫spƒõchu:** {analysis['strategy']['success_probability']:.2%}

### ƒåasov√Ω pl√°n:
"""
        for milestone in analysis['strategy']['timeline']:
            report += f"- {milestone}\n"
        
        return report
