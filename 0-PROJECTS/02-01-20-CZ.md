    async def design_experiment(self, research_topic: str, domain: str) -> Dict[str, Any]:
        """Orchestruje celÃ½ proces designu experimentu"""
        try:

            # UloÅ¾enÃ­ do databÃ¡ze

                "interpretation": interpretation,

                "hypothesis": asdict(hypothesis),
                "design": asdict(design),
                "collection_plan": asdict(collection_plan),
                "analysis_results": analysis_results,

            logger.error(f"Chyba pÅ™i designu experimentu: {str(e)}")
