    async def _analyze_current_content(self, content: str, keywords: List[str]) -> Dict[str, Any]:
        """Analyze current content for SEO factors"""
        try:
            # Word count
            word_count = len(content.split())
            
            # Keyword density
            keyword_density = {}
            content_lower = content.lower()
            
            for keyword in keywords:
                keyword_count = content_lower.count(keyword.lower())
                density = (keyword_count / word_count) * 100 if word_count > 0 else 0
                keyword_density[keyword] = density
            
            # Readability (simplified Flesch score approximation)
            sentences = len([s for s in content.split('.') if s.strip()])
            words = word_count
            readability = 206.835 - (1.015 * (words / max(sentences, 1))) - (84.6 * (self._count_syllables(content) / max(words, 1)))
            
            # Header analysis
            headers = len(re.findall(r'#+\s+', content))  # Markdown headers
            
            analysis = {
                'word_count': word_count,
                'keyword_density': keyword_density,
                'readability_score': max(0, min(100, readability)),
                'header_count': headers,
                'sentences': sentences,
                'avg_sentence_length': words / max(sentences, 1)
            }
            
            return analysis
            
        except Exception as e:
            logger.error(f"Content analysis failed: {e}")
            return {}
    
    async def _optimize_title(self, content: str, keywords: List[str]) -> str:
        """Generate SEO-optimized title"""
        try:
            # Extract main topic from content
            sentences = sent_tokenize(content)
            first_sentence = sentences[0] if sentences else ""
            
            # Use primary keyword in title
            primary_keyword = keywords[0] if keywords else "Guide"
            
            # Generate optimized title
            title_templates = [

                f"Proven {primary_keyword} Techniques That Work"
            ]
            
            optimized_title = random.choice(title_templates)

                optimized_title = optimized_title[:57] + "..."
            
            return optimized_title
            
        except Exception as e:

            return "Optimized Content Title"
    
    async def _count_syllables(self, text: str) -> int:
        """Count syllables in text (simplified)"""
        try:
            # Simple syllable counting heuristic
            vowels = 'aeiouy'
            syllables = 0
            prev_was_vowel = False
            
            for char in text.lower():
                if char in vowels:
                    if not prev_was_vowel:
                        syllables += 1
                    prev_was_vowel = True
                else:
                    prev_was_vowel = False
            
            return max(1, syllables)
            
        except Exception as e:
            return len(text.split())  # Fallback to word count

    async def _analyze_content_tone(self, content: str) -> Dict[str, float]:
        """Analyze the tone characteristics of content"""
        try:
            # Use TextBlob for sentiment analysis
            blob = TextBlob(content)
            sentiment = blob.sentiment
            
            # Calculate tone scores
            tone_scores = {
                'professional': min(1.0, abs(sentiment.polarity) * 0.5 + 0.5),
                'friendly': max(0.0, sentiment.polarity + 0.5),
                'authoritative': min(1.0, (1 - sentiment.subjectivity) * 0.8 + 0.2),
                'engaging': min(1.0, sentiment.subjectivity * 0.7 + 0.3),
                'formal': min(1.0, (1 - sentiment.subjectivity) * 0.6 + 0.2)
            }
            
            return tone_scores
            
        except Exception as e:
            logger.error(f"Tone analysis failed: {e}")
            return {'professional': 0.5, 'friendly': 0.5, 'authoritative': 0.5, 'engaging': 0.5}
