Meteorologick√© informace vƒçetnƒõ teploty, vlhkosti, tlaku, rychlosti vƒõtru a smƒõru vƒõtru.

### Sledov√°n√≠ zneƒçi≈°tƒõn√≠

Dodr≈æov√°n√≠ z√°konn√Ωch limit≈Ø a norem pro kvalitu ovzdu≈°√≠ a emisn√≠ standardy.

class MCPEnvironmentalClient:
    """MCP klient pro environment√°ln√≠ monitoring"""
    
    def __init__(self, base_url: str, api_key: str):
        self.base_url = base_url
        self.api_key = api_key
        self.session = None
    
    async def __aenter__(self):
        self.session = aiohttp.ClientSession()
        return self
    
    async def __aexit__(self, exc_type, exc_val, exc_tb):
        if self.session:
            await self.session.close()
    
    async def get_sensor_data(self, sensor_ids: List[str]) -> List[SensorReading]:
        """Z√≠sk√°n√≠ dat ze senzor≈Ø p≈ôes MCP"""
        headers = {"Authorization": f"Bearer {self.api_key}"}
        
        try:
            async with self.session.get(
                f"{self.base_url}/sensors/readings",
                headers=headers,
                params={"sensor_ids": ",".join(sensor_ids)}
            ) as response:
                data = await response.json()
                
                readings = []
                for item in data.get("readings", []):
                    reading = SensorReading(
                        sensor_id=item["sensor_id"],
                        timestamp=datetime.fromisoformat(item["timestamp"]),
                        pm25=item.get("pm25", 0.0),
                        pm10=item.get("pm10", 0.0),
                        no2=item.get("no2", 0.0),
                        so2=item.get("so2", 0.0),
                        o3=item.get("o3", 0.0),
                        co=item.get("co", 0.0),
                        temperature=item.get("temperature", 0.0),
                        humidity=item.get("humidity", 0.0),
                        pressure=item.get("pressure", 0.0),
                        wind_speed=item.get("wind_speed", 0.0),
                        wind_direction=item.get("wind_direction", 0.0)
                    )
                    readings.append(reading)
                
                return readings
                
        except Exception as e:
            logger.error(f"Chyba p≈ôi z√≠sk√°v√°n√≠ dat ze senzor≈Ø: {e}")
            return []

class AirQualityAnalyzer:
    """Analyz√°tor kvality ovzdu≈°√≠"""
    
    @staticmethod
    def calculate_aqi(pm25: float, pm10: float, no2: float, so2: float, o3: float, co: float) -> int:
        """V√Ωpoƒçet Air Quality Index podle EPA standard≈Ø"""
        
        # Breakpoints pro PM2.5 (24-hour average)
        pm25_breakpoints = [
            (0, 12.0, 0, 50),
            (12.1, 35.4, 51, 100),
            (35.5, 55.4, 101, 150),
            (55.5, 150.4, 151, 200),
            (150.5, 250.4, 201, 300),
            (250.5, 500.4, 301, 500)
        ]
        
        def calculate_individual_aqi(concentration: float, breakpoints: List[tuple]) -> int:
            for bp_lo, bp_hi, aqi_lo, aqi_hi in breakpoints:
                if bp_lo <= concentration <= bp_hi:
                    return int(((aqi_hi - aqi_lo) / (bp_hi - bp_lo)) * (concentration - bp_lo) + aqi_lo)
            return 500  # Hazardous
        
        pm25_aqi = calculate_individual_aqi(pm25, pm25_breakpoints)
        
        # Pro demonstraci pou≈æ√≠v√°me pouze PM2.5, v re√°ln√©m syst√©mu by se pou≈æily v≈°echny parametry
        return pm25_aqi
    
    @staticmethod
    def assess_health_risk(aqi: int) -> Dict[str, str]:
        """Posouzen√≠ zdravotn√≠ho rizika na z√°kladƒõ AQI"""
        
        if aqi <= 50:
            return {
                "level": "Dobr√°",
                "color": "green",
                "description": "Kvalita ovzdu≈°√≠ je uspokojiv√° a zneƒçi≈°tƒõn√≠ p≈ôedstavuje mal√© nebo ≈æ√°dn√© riziko."
            }
        elif aqi <= 100:
            return {
                "level": "M√≠rn√°",
                "color": "yellow",
                "description": "Kvalita ovzdu≈°√≠ je p≈ôijateln√° pro vƒõt≈°inu lid√≠."
            }
        elif aqi <= 150:
            return {
                "level": "Nezdrav√° pro citliv√© skupiny",
                "color": "orange",
                "description": "Citliv√© skupiny mohou poci≈•ovat zdravotn√≠ probl√©my."
            }
        elif aqi <= 200:
            return {
                "level": "Nezdrav√°",
                "color": "red",
                "description": "V≈°ichni mohou zaƒç√≠t poci≈•ovat zdravotn√≠ probl√©my."
            }
        elif aqi <= 300:
            return {
                "level": "Velmi nezdrav√°",
                "color": "purple",
                "description": "Varov√°n√≠ p≈ôed v√°≈æn√Ωmi zdravotn√≠mi probl√©my."
            }
        else:
            return {
                "level": "Nebezpeƒçn√°",
                "color": "maroon",
                "description": "Nouzov√° upozornƒõn√≠ na zdravotn√≠ probl√©my."
            }

class EnvironmentalDatabase:
    """Spr√°va environment√°ln√≠ datab√°ze"""
    
    def __init__(self, database_url: str):
        self.engine = create_engine(database_url)
        Base.metadata.create_all(self.engine)
        Session = sessionmaker(bind=self.engine)
        self.session = Session()
    
    def save_reading(self, reading: SensorReading) -> None:
        """Ulo≈æen√≠ ƒçten√≠ do datab√°ze"""
        
        aqi = AirQualityAnalyzer.calculate_aqi(
            reading.pm25, reading.pm10, reading.no2,
            reading.so2, reading.o3, reading.co
        )
        
        env_data = EnvironmentalData(
            sensor_id=reading.sensor_id,
            timestamp=reading.timestamp,
            pm25=reading.pm25,
            pm10=reading.pm10,
            no2=reading.no2,
            so2=reading.so2,
            o3=reading.o3,
            co=reading.co,
            temperature=reading.temperature,
            humidity=reading.humidity,
            pressure=reading.pressure,
            wind_speed=reading.wind_speed,
            wind_direction=reading.wind_direction,
            aqi=aqi
        )
        
        self.session.add(env_data)
        self.session.commit()
    
    def get_recent_data(self, hours: int = 24) -> List[EnvironmentalData]:
        """Z√≠sk√°n√≠ posledn√≠ch dat"""
        cutoff_time = datetime.now() - timedelta(hours=hours)
        return self.session.query(EnvironmentalData).filter(
            EnvironmentalData.timestamp >= cutoff_time
        ).all()
    
    def get_sensor_trends(self, sensor_id: str, days: int = 7) -> pd.DataFrame:
        """Z√≠sk√°n√≠ trend≈Ø pro konkr√©tn√≠ senzor"""
        cutoff_time = datetime.now() - timedelta(days=days)
        data = self.session.query(EnvironmentalData).filter(
            EnvironmentalData.sensor_id == sensor_id,
            EnvironmentalData.timestamp >= cutoff_time
        ).all()
        
        if not data:
            return pd.DataFrame()
        
        df_data = []
        for record in data:
            df_data.append({
                'timestamp': record.timestamp,
                'pm25': record.pm25,
                'pm10': record.pm10,
                'no2': record.no2,
                'aqi': record.aqi
            })
        
        return pd.DataFrame(df_data)

class ComplianceReporter:
    """Gener√°tor compliance report≈Ø"""
    
    def __init__(self, database: EnvironmentalDatabase):
        self.database = database
        
        # EPA limity (p≈ô√≠klad)
        self.limits = {
            'pm25_daily': 35.0,  # Œºg/m¬≥
            'pm10_daily': 150.0,  # Œºg/m¬≥
            'no2_hourly': 100.0,  # Œºg/m¬≥
            'so2_hourly': 196.0,  # Œºg/m¬≥
        }
    
    def generate_compliance_report(self, days: int = 30) -> Dict[str, Any]:
        """Generov√°n√≠ compliance reportu"""
        
        cutoff_time = datetime.now() - timedelta(days=days)
        data = self.database.session.query(EnvironmentalData).filter(
            EnvironmentalData.timestamp >= cutoff_time
        ).all()
        
        if not data:
            return {"error": "≈Ω√°dn√° data k dispozici"}
        
        # Konverze na DataFrame pro anal√Ωzu
        df_data = []
        for record in data:
            df_data.append({
                'timestamp': record.timestamp,
                'sensor_id': record.sensor_id,
                'pm25': record.pm25,
                'pm10': record.pm10,
                'no2': record.no2,
                'so2': record.so2,
                'aqi': record.aqi
            })
        
        df = pd.DataFrame(df_data)
        
        # Anal√Ωza compliance
        violations = []
        
        # Denn√≠ pr≈Ømƒõry PM2.5
        daily_pm25 = df.groupby([df['timestamp'].dt.date, 'sensor_id'])['pm25'].mean()
        pm25_violations = daily_pm25[daily_pm25 > self.limits['pm25_daily']]
        
        for (date, sensor_id), value in pm25_violations.items():
            violations.append({
                'parameter': 'PM2.5',
                'sensor_id': sensor_id,
                'date': str(date),
                'value': round(value, 2),
                'limit': self.limits['pm25_daily'],
                'exceedance': round(value - self.limits['pm25_daily'], 2)
            })
        
        # Statistiky
        stats = {
            'total_readings': len(df),
            'unique_sensors': df['sensor_id'].nunique(),
            'date_range': {
                'start': df['timestamp'].min().isoformat(),
                'end': df['timestamp'].max().isoformat()
            },
            'average_aqi': round(df['aqi'].mean(), 1),
            'max_aqi': int(df['aqi'].max()),
            'violations_count': len(violations)
        }
        
        return {
            'report_generated': datetime.now().isoformat(),
            'period_days': days,
            'statistics': stats,
            'violations': violations,
            'compliance_rate': round((1 - len(violations) / len(df)) * 100, 2) if len(df) > 0 else 0
        }

class EnvironmentalMonitoringSystem:
    """Hlavn√≠ t≈ô√≠da pro environment√°ln√≠ monitoring"""
    
    def __init__(self, config: Dict[str, Any]):
        self.config = config
        self.database = EnvironmentalDatabase(config['database_url'])
        self.analyzer = AirQualityAnalyzer()
        self.reporter = ComplianceReporter(self.database)
        self.alerts = []
    
    async def collect_sensor_data(self, sensor_ids: List[str]) -> None:
        """Sbƒõr dat ze senzor≈Ø"""
        
        async with MCPEnvironmentalClient(
            self.config['mcp_base_url'],
            self.config['mcp_api_key']
        ) as client:
            readings = await client.get_sensor_data(sensor_ids)
            
            for reading in readings:
                try:
                    # Ulo≈æen√≠ do datab√°ze
                    self.database.save_reading(reading)
                    
                    # Kontrola alert≈Ø
                    aqi = self.analyzer.calculate_aqi(
                        reading.pm25, reading.pm10, reading.no2,
                        reading.so2, reading.o3, reading.co
                    )
                    
                    if aqi > 150:  # Nezdrav√° √∫rove≈à
                        self.generate_alert(reading, aqi)
                    
                    logger.info(f"Zpracov√°no ƒçten√≠ ze senzoru {reading.sensor_id}, AQI: {aqi}")
                    
                except Exception as e:
                    logger.error(f"Chyba p≈ôi zpracov√°n√≠ ƒçten√≠: {e}")
    
    def generate_alert(self, reading: SensorReading, aqi: int) -> None:
        """Generov√°n√≠ alertu pro vysok√© zneƒçi≈°tƒõn√≠"""
        
        health_risk = self.analyzer.assess_health_risk(aqi)
        
        alert = {
            'timestamp': reading.timestamp.isoformat(),
            'sensor_id': reading.sensor_id,
            'aqi': aqi,
            'health_risk': health_risk,
            'pollutants': {
                'pm25': reading.pm25,
                'pm10': reading.pm10,
                'no2': reading.no2
            }
        }
        
        self.alerts.append(alert)
        logger.warning(f"ALERT: Vysok√© zneƒçi≈°tƒõn√≠ na senzoru {reading.sensor_id}, AQI: {aqi}")
    
    def get_current_status(self) -> Dict[str, Any]:
        """Z√≠sk√°n√≠ aktu√°ln√≠ho stavu syst√©mu"""
        
        recent_data = self.database.get_recent_data(hours=1)
        
        if not recent_data:
            return {"status": "≈Ω√°dn√° aktu√°ln√≠ data"}
        
        current_aqi = np.mean([data.aqi for data in recent_data])
        active_sensors = len(set(data.sensor_id for data in recent_data))
        
        return {
            'timestamp': datetime.now().isoformat(),
            'active_sensors': active_sensors,
            'current_aqi': round(current_aqi, 1),
            'health_assessment': self.analyzer.assess_health_risk(int(current_aqi)),
            'recent_alerts': len([alert for alert in self.alerts 
                                if datetime.fromisoformat(alert['timestamp']) > 
                                datetime.now() - timedelta(hours=24)])
        }
