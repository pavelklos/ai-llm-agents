## Key Concepts Explanation

### Multi-Agent Systems (MAS)
Syst√©m v√≠ce nez√°visl√Ωch AI agent≈Ø, kte≈ô√≠ spolupracuj√≠ na ≈ôe≈°en√≠ komplexn√≠ch √∫kol≈Ø. Ka≈æd√Ω agent m√° svou specializaci a m≈Ø≈æe komunikovat s ostatn√≠mi agenty za √∫ƒçelem dosa≈æen√≠ spoleƒçn√©ho c√≠le.

### Agent Specialization

### Natural Language Processing

class TravelPlannerSystem:
    """Hlavn√≠ syst√©m multi-agent travel planneru"""
    
    def __init__(self):
        self.llm = OpenAI(temperature=0.3)
        self.flight_tool = FlightSearchTool()
        self.hotel_tool = HotelSearchTool()
        self.activity_tool = ActivitySearchTool()
        self.memory = UserPreferenceMemory()
        
        self.setup_agents()
    
    def setup_agents(self):
        """Nastaven√≠ specializovan√Ωch agent≈Ø"""
        
        # Flight Agent
        self.flight_agent = Agent(
            role='Flight Specialist',
            goal='Naj√≠t nejlep≈°√≠ leteck√© spojen√≠ podle rozpoƒçtu a preferenc√≠ u≈æivatele',
            backstory="""Jsi expert na leteckou dopravu s p≈ô√≠stupem k aktu√°ln√≠m 
                        dat≈Øm o letech. Specializuje≈° se na hled√°n√≠ optim√°ln√≠ch 
                        spojen√≠ s ohledem na cenu, ƒças a komfort.""",
            verbose=True,
            allow_delegation=False,
            llm=self.llm
        )
        
        # Hotel Agent  
        self.hotel_agent = Agent(
            role='Accommodation Specialist',
            goal='Doporuƒçit ide√°ln√≠ ubytov√°n√≠ podle lokace, rozpoƒçtu a po≈æadavk≈Ø',
            backstory="""Jsi specialista na ubytov√°n√≠ s rozs√°hl√Ωmi znalostmi 
                        hotel≈Ø, penzion≈Ø a alternativn√≠ch forem ubytov√°n√≠. 
                        Rozum√≠≈° lok√°ln√≠m specifik≈Øm a dok√°≈æe≈° naj√≠t nejlep≈°√≠ 
                        pomƒõr ceny a kvality.""",
            verbose=True,
            allow_delegation=False,
            llm=self.llm
        )
        
        # Activity Agent
        self.activity_agent = Agent(
            role='Experience Curator',
            goal='Sestavit program aktivit podle z√°jm≈Ø a dostupn√©ho ƒçasu',
            backstory="""Jsi kur√°tor z√°≈æitk≈Ø s hlubok√Ωmi znalostmi m√≠stn√≠ch 
                        atrakc√≠, kultury a skryt√Ωch klenot≈Ø. Specializuje≈° se 
                        na vytv√°≈ôen√≠ nezapomenuteln√Ωch itiner√°≈ô≈Ø.""",
            verbose=True,
            allow_delegation=False,
            llm=self.llm
        )
        
        # Budget Agent
        self.budget_agent = Agent(
            role='Financial Optimizer',
            goal='Optimalizovat rozpoƒçet a zajistit nejlep≈°√≠ hodnotu za pen√≠ze',
            backstory="""Jsi finanƒçn√≠ poradce specializuj√≠c√≠ se na cestovn√≠ 
                        rozpoƒçty. Dok√°≈æe≈° vyjednat kompromisy mezi r≈Øzn√Ωmi 
                        slo≈ækami cesty a naj√≠t √∫spory bez kompromis≈Ø v kvalitƒõ.""",
            verbose=True,
            allow_delegation=False,
            llm=self.llm
        )
        
        # Coordinator Agent
        self.coordinator_agent = Agent(
            role='Travel Coordinator',
            goal='Koordinovat v≈°echny agenty a sestavit fin√°ln√≠ cestovn√≠ pl√°n',
            backstory="""Jsi hlavn√≠ koordin√°tor s p≈ôehledem o v≈°ech aspektech 
                        cestov√°n√≠. Dok√°≈æe≈° vyv√°≈æit r≈Øzn√© po≈æadavky a sestavit 
                        koherentn√≠ cestovn√≠ pl√°n.""",
            verbose=True,
            allow_delegation=True,
            llm=self.llm
        )
    
    def create_travel_plan(self, travel_request: TravelRequest, user_id: str = "default") -> Dict:
        """Vytvo≈ôen√≠ cestovn√≠ho pl√°nu pomoc√≠ agent≈Ø"""
        
        try:
            # Z√≠sk√°n√≠ u≈æivatelsk√Ωch preferenc√≠
            user_prefs = self.memory.get_preferences(user_id)
            
            # Definice √∫kol≈Ø pro agenty
            flight_task = Task(
                description=f"""
                Najdi nejlep≈°√≠ lety z {travel_request.departure_city} do {travel_request.destination}
                na datum {travel_request.start_date} pro {travel_request.travelers} osob.
                Rozpoƒçet na lety: {travel_request.budget * 0.4} EUR.
                U≈æivatelsk√© preference: {user_prefs}
                """,
                agent=self.flight_agent,
                expected_output="Seznam doporuƒçen√Ωch let≈Ø s cenami a ƒçasy"
            )
            
            hotel_task = Task(
                description=f"""
                Najdi vhodn√© ubytov√°n√≠ v {travel_request.destination} 
                od {travel_request.start_date} do {travel_request.end_date}
                pro {travel_request.travelers} osob.
                Rozpoƒçet na ubytov√°n√≠: {travel_request.budget * 0.35} EUR.
                Preference: {travel_request.preferences}
                """,
                agent=self.hotel_agent,
                expected_output="Seznam doporuƒçen√Ωch hotel≈Ø s cenami a hodnocen√≠m"
            )
            
            activity_task = Task(
                description=f"""
                Navrhni aktivity a program pro pobyt v {travel_request.destination}
                od {travel_request.start_date} do {travel_request.end_date}.
                Rozpoƒçet na aktivity: {travel_request.budget * 0.25} EUR.
                Z√°jmy: {travel_request.preferences.get('interests', 'obecn√©')}
                """,
                agent=self.activity_agent,
                expected_output="Itiner√°≈ô s aktivitami a cenami"
            )
            
            budget_task = Task(
                description=f"""
                Proveƒè rozpoƒçtovou optimalizaci cel√©ho pl√°nu.
                Celkov√Ω rozpoƒçet: {travel_request.budget} EUR.
                Zkontroluj, zda n√°vrhy ostatn√≠ch agent≈Ø se vejdou do rozpoƒçtu
                a navrhni p≈ô√≠padn√© √∫pravy.
                """,
                agent=self.budget_agent,
                expected_output="Rozpoƒçtov√° anal√Ωza a doporuƒçen√© √∫pravy"
            )
            
            coordination_task = Task(
                description=f"""
                Koordinuj v≈°echny agenty a sestav fin√°ln√≠ cestovn√≠ pl√°n.
                Ujisti se, ≈æe v≈°echny slo≈æky jsou kompatibiln√≠ a realistick√©.
                Vytvo≈ô strukturovan√Ω itiner√°≈ô s ƒçasov√Ωm rozpisem.
                """,
                agent=self.coordinator_agent,
                expected_output="Kompletn√≠ cestovn√≠ pl√°n s itiner√°≈ôem"
            )
            
            # Vytvo≈ôen√≠ crew a spu≈°tƒõn√≠ procesu
            crew = Crew(
                agents=[
                    self.flight_agent,
                    self.hotel_agent, 
                    self.activity_agent,
                    self.budget_agent,
                    self.coordinator_agent
                ],
                tasks=[flight_task, hotel_task, activity_task, budget_task, coordination_task],
                process=Process.sequential,
                verbose=True
            )
            
            # Spu≈°tƒõn√≠ pl√°nov√°n√≠
            result = crew.kickoff()
            
            # Simulace skuteƒçn√Ωch dat pro demonstraci
            mock_plan = self._generate_mock_plan(travel_request)
            
            return {
                "status": "success",
                "travel_plan": mock_plan,
                "agent_result": str(result),
                "total_cost": mock_plan["budget"]["total_cost"]
            }
            
        except Exception as e:
            return {
                "status": "error",
                "message": f"Chyba p≈ôi vytv√°≈ôen√≠ cestovn√≠ho pl√°nu: {str(e)}"

            request.departure_city, 
            request.destination, 
            request.start_date, 
            request.budget * 0.4
        )
        
        hotels = self.hotel_tool.search_hotels(
            request.destination,
            request.start_date,
            request.end_date, 
            request.budget * 0.35 / 3  # 3 noci
        )
        
        activities = self.activity_tool.search_activities(
            request.destination,
            request.preferences.get('interests', '').split(','),
            request.budget * 0.25
        )
        
        return {
            "destination": request.destination,
            "dates": {
                "start": request.start_date,
                "end": request.end_date
            },
            "flights": {
                "outbound": flights[0] if flights else None,
                "return": flights[1] if len(flights) > 1 else flights[0] if flights else None
            },
            "accommodation": hotels[0] if hotels else None,
            "activities": activities[:3] if activities else [],
            "budget": {
                "total_budget": request.budget,
                "flights_cost": flights[0]["price"] * 2 if flights else 0,
                "hotel_cost": hotels[0]["price_per_night"] * 3 if hotels else 0,
                "activities_cost": sum([a["price"] for a in activities[:3]]) if activities else 0,
                "total_cost": (flights[0]["price"] * 2 if flights else 0) +
