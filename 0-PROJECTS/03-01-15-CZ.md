class SteamAPIClient:
    def __init__(self, api_key: str):
        self.api_key = api_key
        self.base_url = "https://api.steampowered.com"
    
    async def get_player_stats(self, steam_id: str, game_id: str) -> Dict:
        """Z√≠sk√°n√≠ statistik hr√°ƒçe ze Steam API"""
        try:
            url = f"{self.base_url}/ISteamUserStats/GetPlayerAchievements/v0001/"
            params = {
                "key": self.api_key,
                "steamid": steam_id,
                "appid": game_id
            }
            response = requests.get(url, params=params)
            return response.json() if response.status_code == 200 else {}
        except Exception as e:
            st.error(f"Chyba p≈ôi naƒç√≠t√°n√≠ Steam dat: {e}")
            return {}
    
    async def get_game_achievements(self, game_id: str) -> List[Dict]:
        """Z√≠sk√°n√≠ seznamu achievement≈Ø pro hru"""
        try:
            url = f"{self.base_url}/ISteamUserStats/GetSchemaForGame/v2/"
            params = {
                "key": self.api_key,
                "appid": game_id
            }
            response = requests.get(url, params=params)
            data = response.json()
            return data.get("game", {}).get("availableGameStats", {}).get("achievements", [])
        except Exception as e:
            st.error(f"Chyba p≈ôi naƒç√≠t√°n√≠ achievement≈Ø: {e}")
            return []

class TwitchAPIClient:
    def __init__(self, client_id: str, client_secret: str):
        self.client_id = client_id
        self.client_secret = client_secret
        self.access_token = None
        self.base_url = "https://api.twitch.tv/helix"
    
    async def authenticate(self):
        """Autentifikace s Twitch API"""
        try:
            url = "https://id.twitch.tv/oauth2/token"
            data = {
                "client_id": self.client_id,
                "client_secret": self.client_secret,
                "grant_type": "client_credentials"
            }
            response = requests.post(url, data=data)
            if response.status_code == 200:
                self.access_token = response.json()["access_token"]
        except Exception as e:
            st.error(f"Chyba p≈ôi autentifikaci Twitch: {e}")
    
    async def get_game_streams(self, game_name: str, limit: int = 10) -> List[Dict]:
        """Z√≠sk√°n√≠ stream≈Ø pro konkr√©tn√≠ hru"""
        if not self.access_token:
            await self.authenticate()
        
        try:
            headers = {
                "Client-ID": self.client_id,
                "Authorization": f"Bearer {self.access_token}"
            }
            url = f"{self.base_url}/streams"
            params = {
                "game_name": game_name,
                "first": limit
            }
            response = requests.get(url, headers=headers, params=params)
            return response.json().get("data", []) if response.status_code == 200 else []
        except Exception as e:
            st.error(f"Chyba p≈ôi naƒç√≠t√°n√≠ stream≈Ø: {e}")
            return []

class GameKnowledgeBase:
    def __init__(self, db_path: str = "game_knowledge.db"):
        self.db_path = db_path
        self.embeddings = OpenAIEmbeddings()
        self.vectorstore = None
        self.qa_chain = None
        self._init_database()
        self._init_vectorstore()
    
    def _init_database(self):
        """Inicializace SQLite datab√°ze"""
        conn = sqlite3.connect(self.db_path)
        cursor = conn.cursor()
        
        # Tabulka pro hern√≠ pr≈Øvodce
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS game_guides (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                game_id TEXT NOT NULL,
                title TEXT NOT NULL,
                content TEXT NOT NULL,
                category TEXT NOT NULL,
                difficulty TEXT NOT NULL,
                tags TEXT NOT NULL,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)
        
        # Tabulka pro achievementy
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS achievements (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                game_id TEXT NOT NULL,
                achievement_id TEXT NOT NULL,
                name TEXT NOT NULL,
                description TEXT NOT NULL,
                unlock_guide TEXT NOT NULL,
                rarity REAL NOT NULL,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)
        
        conn.commit()
        conn.close()
    
    def _init_vectorstore(self):
        """Inicializace vektorov√©ho √∫lo≈æi≈°tƒõ"""
        try:
            self.vectorstore = Chroma(
                persist_directory="./chroma_db",
                embedding_function=self.embeddings
            )
            
            llm = OpenAI(temperature=0.1)
            self.qa_chain = RetrievalQA.from_chain_type(
                llm=llm,
                chain_type="stuff",
                retriever=self.vectorstore.as_retriever(search_kwargs={"k": 5})
            )
        except Exception as e:
            st.error(f"Chyba p≈ôi inicializaci vektorov√©ho √∫lo≈æi≈°tƒõ: {e}")
    
    def add_game_guide(self, guide: GameGuide):
        """P≈ôid√°n√≠ hern√≠ho pr≈Øvodce do datab√°ze"""
        try:
            conn = sqlite3.connect(self.db_path)
            cursor = conn.cursor()
            
            cursor.execute("""
                INSERT INTO game_guides (game_id, title, content, category, difficulty, tags)
                VALUES (?, ?, ?, ?, ?, ?)
            """, (
                guide.game_id,
                guide.title,
                guide.content,
                guide.category,
                guide.difficulty,
                json.dumps(guide.tags)
            ))
            
            conn.commit()
            conn.close()
            
            # P≈ôid√°n√≠ do vektorov√©ho √∫lo≈æi≈°tƒõ
            if self.vectorstore:
                self.vectorstore.add_texts(
                    texts=[f"{guide.title}: {guide.content}"],
                    metadatas=[{
                        "game_id": guide.game_id,
                        "category": guide.category,
                        "difficulty": guide.difficulty
                    }]
                )
        except Exception as e:
            st.error(f"Chyba p≈ôi p≈ôid√°v√°n√≠ pr≈Øvodce: {e}")
    
    def search_guides(self, query: str, game_id: Optional[str] = None) -> str:
        """Vyhled√°n√≠ relevantn√≠ch pr≈Øvodc≈Ø pomoc√≠ RAG"""
        try:
            if self.qa_chain:
                if game_id:
                    query = f"Pro hru {game_id}: {query}"
                
                response = self.qa_chain.run(query)
                return response
            return "Syst√©m nen√≠ dostupn√Ω"
        except Exception as e:
            st.error(f"Chyba p≈ôi vyhled√°v√°n√≠: {e}")
            return "Chyba p≈ôi vyhled√°v√°n√≠ informac√≠"
    
    def get_achievement_guide(self, game_id: str, achievement_name: str) -> Optional[str]:
        """Z√≠sk√°n√≠ pr≈Øvodce pro konkr√©tn√≠ achievement"""
        try:
            conn = sqlite3.connect(self.db_path)
            cursor = conn.cursor()
            
            cursor.execute("""
                SELECT unlock_guide FROM achievements 
                WHERE game_id = ? AND name LIKE ?
            """, (game_id, f"%{achievement_name}%"))
            
            result = cursor.fetchone()
            conn.close()
            
            return result[0] if result else None
        except Exception as e:
            st.error(f"Chyba p≈ôi naƒç√≠t√°n√≠ achievement pr≈Øvodce: {e}")
            return None

    async def get_personalized_strategy(self, game_id: str, player_query: str, player_stats: Optional[PlayerStats] = None) -> Dict[str, Any]:
        """Z√≠sk√°n√≠ personalizovan√© strategie pro hr√°ƒçe"""
        try:
            # Z√°kladn√≠ vyhled√°n√≠ v knowledge base
            base_response = self.knowledge_base.search_guides(player_query, game_id)
            
            # P≈ôid√°n√≠ kontextu podle statistik hr√°ƒçe
            context = ""
            if player_stats:
                if player_stats.level < 10:
                    context = "Jako zaƒç√°teƒçn√≠k se zamƒõ≈ô na z√°klady. "

                    context = "Jako pokroƒçil√Ω hr√°ƒç m≈Ø≈æe≈° zkusit slo≈æitƒõj≈°√≠ strategie. "
                else:
                    context = "Jako zku≈°en√Ω hr√°ƒç m≈Ø≈æe≈° experimentovat s pokroƒçil√Ωmi taktikami. "
            
            # Z√≠sk√°n√≠ souvisej√≠c√≠ch stream≈Ø
            streams = await self.twitch_client.get_game_streams(f"game_{game_id}", 3)
            
            return {
                "strategy": context + base_response,
                "related_streams": streams,
                "confidence": 0.85,
                "sources": ["knowledge_base", "community"]
            }
        except Exception as e:
            st.error(f"Chyba p≈ôi generov√°n√≠ strategie: {e}")
            return {"strategy": "Omlouv√°me se, strategie nen√≠ dostupn√°", "confidence": 0}
    
    async def get_achievement_progress(self, steam_id: str, game_id: str) -> List[Dict]:
        """Z√≠sk√°n√≠ pokroku v achievementech s tipy"""
        try:
            # Z√≠sk√°n√≠ dat ze Steam API
            player_achievements = await self.steam_client.get_player_stats(steam_id, game_id)
            game_achievements = await self.steam_client.get_game_achievements(game_id)
            
            progress_data = []
            for achievement in game_achievements:
                achievement_id = achievement.get("name", "")
                unlocked = False
                
                # Kontrola, zda je achievement odemknut√Ω
                if "playerstats" in player_achievements:
                    for player_ach in player_achievements["playerstats"].get("achievements", []):
                        if player_ach.get("apiname") == achievement_id:
                            unlocked = player_ach.get("achieved", 0) == 1
                            break
                
                # Z√≠sk√°n√≠ pr≈Øvodce pro achievement
                guide = self.knowledge_base.get_achievement_guide(game_id, achievement.get("displayName", ""))
                
                progress_data.append({
                    "name": achievement.get("displayName", ""),
                    "description": achievement.get("description", ""),
                    "unlocked": unlocked,
                    "guide": guide or "Pr≈Øvodce nen√≠ dostupn√Ω",
                    "icon": achievement.get("icon", "")
                })
            
            return progress_data
        except Exception as e:
            st.error(f"Chyba p≈ôi naƒç√≠t√°n√≠ achievement≈Ø: {e}")
            return []
