## 2. Vysvƒõtlen√≠ Kl√≠ƒçov√Ωch Koncept≈Ø

### Multi-Agent Systems (Syst√©my v√≠ce agent≈Ø)
Distribuovan√Ω syst√©m slo≈æen√Ω z autonomn√≠ch AI agent≈Ø, kte≈ô√≠ spolupracuj√≠ na ≈ôe≈°en√≠ komplexn√≠ch √∫kol≈Ø. Ka≈æd√Ω agent m√° specifickou roli a expertizu.

### Ingredient Analysis (Anal√Ωza slo≈æen√≠)
Proces identifikace a anal√Ωzy jednotliv√Ωch slo≈æek potravin, vƒçetnƒõ jejich nutriƒçn√≠ch hodnot a potenci√°ln√≠ch rizik.

### Allergen Detection (Detekce alergen≈Ø)
Automatick√° identifikace alergick√Ωch l√°tek v potravin√°ch a upozornƒõn√≠ na potenci√°ln√≠ rizika pro alergiky.

### Nutritional Optimization (Optimalizace v√Ω≈æivy)
Proces vylep≈°ov√°n√≠ nutriƒçn√≠ho profilu j√≠del podle specifick√Ωch dietn√≠ch c√≠l≈Ø a zdravotn√≠ch pot≈ôeb.

### Recipe Generation (Generov√°n√≠ recept≈Ø)

### Dietary Compliance (Dodr≈æov√°n√≠ diety)
Zaji≈°tƒõn√≠, ≈æe navrhovan√© recepty a j√≠dla odpov√≠daj√≠ specifick√Ωm dietn√≠m po≈æadavk≈Øm (veg√°nsk√°, bezlepkov√°, diabetick√° dieta atd.).

class DietaryRestriction(Enum):

    DIABETIC = "diabetic"

class AllergenType(Enum):
    GLUTEN = "gluten"
    DAIRY = "dairy"
    NUTS = "nuts"
    EGGS = "eggs"
    SOY = "soy"
    FISH = "fish"

# N√°stroje pro agenty
class IngredientAnalysisTool(BaseTool):
    name = "ingredient_analysis"
    description = "Analyzuje slo≈æen√≠ ingredienc√≠ a jejich nutriƒçn√≠ hodnoty"
    
    def _run(self, ingredient_list: str) -> str:
        # Simulace anal√Ωzy ingredienc√≠
        ingredients = ingredient_list.split(',')
        analysis_results = []
        
        for ingredient in ingredients:
            ingredient = ingredient.strip()
            # Simulovan√© nutriƒçn√≠ hodnoty
            nutritional_info = NutritionalInfo(
                calories=100.0,
                protein=5.0,
                carbs=15.0,
                fat=3.0,
                fiber=2.0,
                sodium=200.0,
                sugar=8.0
            )
            
            # Detekce potenci√°ln√≠ch alergen≈Ø
            allergens = []
            if any(word in ingredient.lower() for word in ['ml√©ko', 's√Ωr', 'm√°slo']):
                allergens.append(AllergenType.DAIRY)
            if any(word in ingredient.lower() for word in ['mouka', 'p≈°enice']):
                allergens.append(AllergenType.GLUTEN)
            if any(word in ingredient.lower() for word in ['o≈ôech', 'mandl']):
                allergens.append(AllergenType.NUTS)
                
            analysis_results.append({
                'ingredient': ingredient,
                'allergens': [a.value for a in allergens],
                'nutrition': nutritional_info.__dict__
            })
        
        return json.dumps(analysis_results, ensure_ascii=False)

class AllergenDetectionTool(BaseTool):
    name = "allergen_detection"
    description = "Detekuje alergeny v seznamu ingredienc√≠"
    
    def _run(self, ingredients_analysis: str) -> str:
        data = json.loads(ingredients_analysis)
        allergen_warnings = []
        
        for item in data:
            if item['allergens']:
                allergen_warnings.append({
                    'ingredient': item['ingredient'],
                    'detected_allergens': item['allergens'],
                    'warning_level': 'high' if len(item['allergens']) > 1 else 'medium'
                })
        
        return json.dumps(allergen_warnings, ensure_ascii=False)

class NutritionalOptimizationTool(BaseTool):
    name = "nutritional_optimization"
    description = "Optimalizuje nutriƒçn√≠ profil receptu"
    
    def _run(self, recipe_nutrition: str, user_goals: str) -> str:
        nutrition_data = json.loads(recipe_nutrition)
        goals = json.loads(user_goals)
        
        recommendations = []
        total_calories = sum(item['nutrition']['calories'] for item in nutrition_data)
        
        if total_calories > goals.get('daily_calorie_goal', 2000) * 0.3:
            recommendations.append("Sni≈æte porce nebo nahraƒète kalorick√© ingredience")
        
        total_sodium = sum(item['nutrition']['sodium'] for item in nutrition_data)
        if total_sodium > 800:  # Vysok√Ω obsah sod√≠ku
            recommendations.append("Sni≈æte obsah soli nebo pou≈æijte byliny pro ochucen√≠")
            
        return json.dumps({
            'total_calories': total_calories,
            'total_sodium': total_sodium,
            'recommendations': recommendations
        }, ensure_ascii=False)

class RecipeGenerationTool(BaseTool):

    description = "Generuje recepty na z√°kladƒõ ingredienc√≠ a omezen√≠"
    
    def _run(self, available_ingredients: str, dietary_restrictions: str) -> str:

        restrictions = dietary_restrictions.split(',')
        
        # Simulace generov√°n√≠ receptu

            'cook_time': '20 minut',

    return ingredient_analyst, allergen_detector, nutrition_optimizer, recipe_generator

# Definice √∫kol≈Ø
def create_tasks(user_profile: UserProfile, ingredients: List[str]):
    ingredient_list = ','.join(ingredients)
    user_restrictions = ','.join([r.value for r in user_profile.dietary_restrictions])
    user_allergies = ','.join([a.value for a in user_profile.allergies])
    
    # √ökol 1: Anal√Ωza ingredienc√≠
    analyze_ingredients_task = Task(
        description=f'Analyzuj n√°sleduj√≠c√≠ ingredience: {ingredient_list}. '
                   f'Urƒçi jejich nutriƒçn√≠ hodnoty a potenci√°ln√≠ alergeny.',
        agent=None,  # Bude p≈ôi≈ôazen pozdƒõji
        expected_output='JSON s detailn√≠ anal√Ωzou ka≈æd√© ingredience vƒçetnƒõ nutriƒçn√≠ch hodnot a alergen≈Ø'
    )
    
    # √ökol 2: Detekce alergen≈Ø
    detect_allergens_task = Task(
        description=f'Na z√°kladƒõ anal√Ωzy ingredienc√≠ zkontroluj p≈ô√≠tomnost alergen≈Ø. '
                   f'U≈æivatel je alergick√Ω na: {user_allergies}',
        agent=None,
        expected_output='Seznam detekovan√Ωch alergen≈Ø s √∫rovn√≠ rizika'
    )
    
    # √ökol 3: Optimalizace v√Ω≈æivy
    optimize_nutrition_task = Task(
        description=f'Optimalizuj nutriƒçn√≠ profil pro u≈æivatele s c√≠lem {user_profile.daily_calorie_goal} kalori√≠ dennƒõ. '
                   f'Dietn√≠ omezen√≠: {user_restrictions}',
        agent=None,
        expected_output='Doporuƒçen√≠ pro optimalizaci nutriƒçn√≠ho profilu'
    )
    
    # √ökol 4: Generov√°n√≠ receptu
    generate_recipe_task = Task(
        description=f'Vytvo≈ô recept z dostupn√Ωch ingredienc√≠: {ingredient_list}. '
                   f'Respektuj dietn√≠ omezen√≠: {user_restrictions} a alergie: {user_allergies}',
        agent=None,
        expected_output='Kompletn√≠ recept s postupem p≈ô√≠pravy a dietn√≠mi informacemi'
    )
    
    return analyze_ingredients_task, detect_allergens_task, optimize_nutrition_task, generate_recipe_task

# Hlavn√≠ aplikace
class FoodSafetyPlatform:
    def __init__(self):
        self.agents = create_agents()
        
    def process_food_request(self, user_profile: UserProfile, ingredients: List[str]) -> Dict[str, Any]:
        try:
            # Vytvo≈ôen√≠ √∫kol≈Ø
            tasks = create_tasks(user_profile, ingredients)
            
            # P≈ôi≈ôazen√≠ agent≈Ø k √∫kol≈Øm
            tasks[0].agent = self.agents[0]  # ingredient_analyst
            tasks[1].agent = self.agents[1]  # allergen_detector
            tasks[2].agent = self.agents[2]  # nutrition_optimizer
            tasks[3].agent = self.agents[3]  # recipe_generator
            
            # Vytvo≈ôen√≠ t√Ωmu
            crew = Crew(
                agents=list(self.agents),
                tasks=list(tasks),
                process=Process.sequential,
                verbose=True
            )
            
            # Spu≈°tƒõn√≠ anal√Ωzy
            result = crew.kickoff()
            
            return {
                'status': 'success',
                'analysis_result': str(result),
                'timestamp': datetime.now().isoformat(),
                'user_profile': user_profile.dict()
            }
            
        except Exception as e:
            return {
                'status': 'error',
                'error_message': str(e),
                'timestamp': datetime.now().isoformat()
            }
